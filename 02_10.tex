\section*{Expected Value of Functions of Random Variables}

\par\noindent
Suppose $X_{1},\ldots,X_{n}$ are random variables, and $Y=g(X_{1},\ldots,X_{n})$. Then
\begin{itemize}
	\item Discrete case: For pmf $p(x_{1},\ldots,x_{n})$, we have $\displaystyle E(Y)=\sum_{\mathclap{x_{i}\in{}X_{i}}}g(x_{1},\ldots,x_{n})p(x_{1},\ldots,x_{n})$, if $E(\abs{g(X_{1},\ldots,X_{n})})$ is finite.
	\item Continuous case: For pdf $f(x_{1},\ldots,x_{n})$, we have $\displaystyle E(Y)=\int_{X_{i}}g(x_{1},\ldots,x_{n})p(x_{1},\ldots,x_{n})$, if $E(\abs{g(X_{1},\ldots,X_{n})})$ is finite.
\end{itemize}

\ex{
	If $E(X_{i})=\mu$, for $i=1,\ldots,n$, then $E(X_{1}+\cdots+X_{n})=n\mu$, and $E(\bar{X})=E((X_{1}+\cdots+X_{n})/n)=\mu$.\n
}

\par\noindent
If $X,Y$ are independent random variables, and $g,h$ functions, then $E(g(X)h(X))=E(g(X))E(h(X))$; $E(XY)=E(X)E(Y)$.\break

\ex{
	Suppose $X_{1}\sim\Normal(\mu_{1},\sigma_{1}^{2})$ and $X_{2}\sim\Normal(\mu_{2},\sigma_{2}^{2})$, with $X_{1}\perp{}X_{2}$. Then $E(X_{1}X_{2})=E(X_{1})E(X_{2})=\mu_{1}\mu_{2}$.\n
}

\subsection*{Variance and Standard Deviation}

\defn{
	Let $X$ be a random variable with mean $\mu$. Then the \u{variance} is
	\[
		\Var(X)=E[(X-\mu)^{2}]
	\]
	We denote the \u{standard deviation} of $X$ as $\sigma_{X}$, where $\Var(X)=\sigma_{X}^{2}$, i.e., $\sqrt{\Var(X)}=\sigma_{X}$.\n
}

\par\noindent
Properties:
\begin{itemize}
	\item $\Var(X)=E[(X-\mu)^{2}]=E(X^{2}-2\mu{}X+\mu^{2})=E(X^{2})-2\mu^{2}+\mu^{2}=E(X^{2})-\mu^{2}$
	\item $\Var(a+bX)=E((a+bX-a-b\mu)^{2})=E(b^{2}(X-\mu)^{2})=b^{2}\Var(X)$.
\end{itemize}

\exer{
	Derive the following:
	\begin{itemize}
		\item $X\sim\Ber(p)\rightsquigarrow{}\Var(X)=p(1-p)$
		\item $X\sim\Bin(n,p)\rightsquigarrow{}\Var(X)=np(1-p)$
		\item $X\sim\Geom(p)\rightsquigarrow{}\Var(X)=(1-p)/p^{2}$
		\item $X\sim\Poisson(\lambda)\rightsquigarrow{}\Var(X)=\lambda$
		\item $X\sim\Unif(a,b)\rightsquigarrow{}\Var(X)=(b-a)^{2}/12$
		\item $X\sim\Exp(\lambda)\rightsquigarrow{}\Var(X)=1/\lambda^{2}$
		\item $X\sim\DistGamma(\lambda,\alpha)\rightsquigarrow{}\Var(X)=\alpha/\lambda^{2}$
		\item $X\sim\Beta(a,b)\rightsquigarrow{}\Var(X)=\frac{ab}{(a+b)^{2}(a+b+1)}$
		\item $X\sim\Normal(\mu,\sigma^{2})\rightsquigarrow{}\Var(X)=\sigma^{2}$
	\end{itemize}
}

\par\noindent
The standard deviation measures how ``spread out'' the distribution is about the mean. For the normal distribution, $\mu\pm{}2\sigma$ and $\mu\pm{}3\sigma$ covers $95.44\%$ and $99.74\%$ of the mass of the distribution.\n

\thm{
	(Chebychev Inequality) Say $X$ is a random variable, with $E(X)=\mu$ and $|Var(X)=\sigma^{2}$.\n
	For $k>0$, $P(\abs{X-\mu}\ge{}k)\le\sigma^{2}/k^{2}$, or equivalently, $P(\abs{X-\mu}\ge{}k\sigma)\le{}1/k^{2}$.\nn
	Proof: Let $R+\set{x:\abs{x-\mu}\ge{}k}$. Then
	\[
		P(\abs{X-\mu}\ge{}k)=\int_{R}1\cdot{}f(x)\,dx\le\int_{R}\frac{(x-\mu)^{2}}{k^{2}}f(x)\,dx=\frac{1}{k^{2}}\int_{R}(x-\mu)^{2}f(x)\,dx=\frac{\sigma^{2}}{k^{2}}
	\]
	\proven
}
