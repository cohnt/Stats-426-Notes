\subsection*{Central Limit Theorem}

\thm{
	Let $X_{1},X_{2},\ldots$ be a sequence of independent random variables, with mean $0$ and variance $\sigma^{2}$, common cdf $F$, and common mgf $M$ defined in a neighborhood of $0$. Let $S_{n}=\sum_{i=1}^{n}X_{i}$. Then
	\[
		\lim_{n\to\infty}P\paren{\frac{S_{n}}{\sigma\sqrt{n}}\le{}x}=\phi(x),\;-\infty<x<\infty
	\]
	\up\nn
	Proof: We know $E(S_{n})=n\mu$, $\Var(S_{n})=n\sigma^{2}$. Suppose that $E(X)=\mu$. Let $Z_{n}=\frac{S_{n}-n\mu}{\sigma\sqrt{n}}=\frac{\bar{X}_{n}-\mu}{\sigma/\sqrt{n}}$. Then $Z_{n}\overset{D}{\to}\Normal(0,1)$. That is, for large $n$, the distribution of $S_{n}$ is approximately $\Normal(n\mu=0,n\sigma^{2})$. So $\bar{X}_{n}$ is approximately $\Normal(\mu,\sigma^{2}/n)$.\proven
}

\ex{
	Suppose $Y_{n}\sim\Bin(n,p)$, where $p\in(0,1)$. Then
	\[
		\frac{Y_{n}-np}{\sqrt{np(1-p)}}\overset{D}{\to}\Normal(0,1)
	\]
	Or $Y_{n}$ is approximately $\Normal(np,np(1-p))$. Our rule of thumb here is that he normal approximation of the CLT is valid when $np\ge{}5$ and $n(1-p)\ge{}5$. The closer $p$ is to $0$ or $1$, the larger the sample size must be.\n
}

\ex{
	Suppose $Y_{n}\sim\Poisson(\lambda_{n})$, where $\lambda_{n}\to\infty$. Then
	\[
		\frac{Y_{n}-\lambda_{n}}{\sqrt{\lambda_{n}}}\overset{D}{\to}\Normal(0,1)
	\]
	Or $Y_{n}$ is approximately $\Normal(\lambda_{n},\lambda_{n})$.\n
}

\ex{
	A coin is tossed $100$ times, and $60$ are heads. Is it fair?\nn
	Well, let $X$ be the number of heads in $100$ tosses. $E(X)=np=50$, $\Var(X)=np(1-p)=25$ if the coin is fair. Assume the coin is fair. Then
	\[
		P(X\ge{}60)=P\paren{\frac{X-50}{5}\ge\frac{60-50}{5}}\approx1-\phi(2)\approx0.0228
	\]
	So it's probably not a fair coin!\n
}

\section*{Distributions Derived from the Normal Distribution}

\defn{
	The \u{chi-square distribution} with $n$ degrees of freedom is defined as follows: $Z\sim\Normal(0,1)$, $U=Z^{2}\sim\chi_{1}^{2}$. If $U_{1},\ldots,U_{n}$ are independent chi-square distributions with $1$ degree of freedom, then their sum is a chi-square distribution with $n$ degrees of freedom, i.e., $Y=U_{1}+\cdots+U_{n}\sim\chi_{n}^{2}$.\n
}

\defn{
	The \u{$t$-distribution} with $n$ degrees of freedom is defined as follows: $Z\sim\Normal(0,1)$, $U\sim\chi_{n}^{2}$, $Z\perp{}U$, then $T=\frac{Z}{\sqrt{U/n}}$ has a $t_{n}$-distribution.\n
}

\defn{
	The \u{$F$-distribution} with $m$ and $n$ degrees of freedom is defined as follows: $U\sim\chi_{n}^{2}$, $V\sim\chi_{m}^{2}$, $U\perp{}V$. $F=\frac{V/m}{U/n}$ has a $F_{m,n}$ distribution: $F\sim{}F_{m,n}$.\n
}

\subsection*{Chi-Square Distribution}

\begin{itemize}
	\item $Y\sim\chi_{n}^{2}\equiv{}Y\sim\DistGamma(\lambda=1/2,\alpha=n/2)$
	\item $f_{Y}(y)=\frac{1}{2^{n/2}\Gamma(n/2)}y^{(n/2)-1}e^{-(1/2)y}$
	\item $E(Y)=\frac{\alpha}{\lambda}=\frac{n/2}{1/2}=n$
	\item $\Var(Y)=\frac{\alpha}{\lambda^{2}}=\frac{n/2}{1/4}=2n$
	\item The mgf for $\DistGamma(\lambda,\alpha)$ is $(1-t/\lambda)^{-\alpha}$ for $t<\lambda$, so the mgf for $\chi_{n}^{2}$ is
	\[
		(1-2t)^{-n/2}=\underbrace{(1-2t)^{-1/2}\cdots(1-2t)^{-1/2}}_{n}
	\]
\end{itemize}

\exer{
	Suppose $Z_{1},\ldots,Z_{N}\sim\Normal(0,1)$ are i.i.d. Then $Z_{1}+\cdots+Z_{n}\sim\chi_{n}^{2}$.\nn
	Prove by induction: For $n=1$, use transformation or the cdf. Note that $U_{1}\sim\DistGamma(\lambda,\alpha_{1})$ and $U_{2}\sim\DistGamma(\lambda,\alpha_{2})$, with $U_{1}\perp{}U_{2}$, implies that $U_{1}+U_{2}\sim\DistGamma(\lambda,\alpha_{1}+\alpha_{2})$.\n
}

\par\noindent
What about the CLT? Well, if $U_{i}=Z_{i}^{2}$, where $Z_{i}\sim\Normal(0,1)$ i.i.d., then $U_{i}\sim\chi^{2}_{1}$, or $\DistGamma(1/2,1/2)$.\n
Let $S_{n}=U_{1}+\cdots+U_{n}$. Then CLT implies that $S_{n}$ is approximately normal, but we need to find its mean and variance.
\[
	E(S_{n})=nE(U_{1})=n\cdot{}1=n\qquad\Var(S_{n})=n\Var(U_{1})=n\cdot{}2=2n
\]
Thus, $S_{n}$ is approximately $\Normal(n,2n)$.\n
