\defn{
	The \u{multinomial distribution} is a multivariate version of the binomial distribution, where there are $n$ independent trials, and each trial results in $1$ of $r$ different outcomes, with probabilities $p_{1},\ldots,p_{r}$ \st{} $p_{1}+\cdots+p_{r}=1$. If $(X_{1},\ldots,X_{r})$ is a vector of binomial random variables, then the pmf of the multinomial distribution is
	\[
		p(n_{1},\ldots,n_{r})=\binom{n}{n_{1},\ldots,n_{r}}p_{1}^{n_{1}}\cdots{}p_{r}^{n_{r}}
	\]
	when the $n_{i}$'s sum to $n$, and $0$ otherwise. Note that the pmf is defined in terms of the \u{multinomial coefficient}
	\[
		\binom{n}{n_{1},\ldots,n_{r}}\eqdef\frac{n!}{n_{1}!\cdots{}n_{r}!}
	\]
}

\defn{
	Let $X$ and $Y$ be continuous random variables. Then $f(x,y)$ is the \u{joint probability density function} of $(X,Y)$ if, for any $A\subseteq\R^{2}$,
	\[
		P((x,y)\in{}A)=\iint_{A}f(x,y)\,dx\,dy
	\]
	This joint pdf must satisfy
	\begin{itemize}[leftmargin=4\parindent]
		\item $f(x,y)\ge{}0$, $\forall(x,y)$
		\item $\displaystyle\iint_{\R^{2}}f(x,y)\,dx\,dy$
	\end{itemize}
}

\defn{
	The \u{marginal probability distrbutions} of $X$ and $Y$ are
	\[
		f_{X}(x)=\int_{\R}f(x,y)\,dy\qquad{}f_{Y}(y)=\int_{\R}f(x,y)\,dx
	\]
	(respectively).\n
}

\defn{
	$X$ and $Y$ are \u{independent} if, $\forall(x,y)$, $f(x,y)=f_{X}(x)f_{Y}(y)$.\n
}

\defn{
	The \u{joint cumulative distribution function} is
	\[
		F(x,y)=P(X\le{}x,Y\le{}y)=\int_{-\infty}^{x}\int_{-\infty}^{y}f(u,v)\,dv\,du
	\]
}

\ex{
	Let $(X,Y)$ have the joint pdf
	\[
		f(x,y)=\left\{\begin{array}{ll}
			\frac{4}{5}xy & 0\le{}x\le{}1,2\le{}y\le{}3\\
			0 & \ptxt{otherwise}
		\end{array}\right.
	\]
	Then the marginal pdfs are
	\begin{align*}
		f_{X}(x) & =\int_{2}^{3}\frac{4}{5}xy\,dy=\brack{\frac{2}{5}xy^{2}}_{y=2}^{y=3}=\frac{2}{5}x(9-4)=2x\qquad\ptxt{(for $0\le{}x\le{}1$)}\\
		f_{Y}(y) & =\int_{0}^{1}\frac{4}{5}xy\,dx=\brack{\frac{2}{5}x^{2}y}_{x=0}^{x=1}=\frac{2}{5}y(1-0)=\frac{2}{5}y\qquad\!\ptxt{(for $2\le{}y\le{}3$)}
	\end{align*}
	Are $X$ and $Y$ independent? Yes! $f_{X}(x)f_{Y}(y)=\frac{4}{5}xy=f(x,y)$, $\forall{}(x,y)$. We write ``$X\perp{}Y$''.\n
}

\ex{
	A point is chosen randomly in a disc of radius $1$. Let $X,Y$ be its $x,y$ coordinates. The pdf of $X,Y$ is
	\[
		f(x,y)=\left\{\begin{array}{ll}
			\frac{1}{\pi} & x^{2}+y^{2}\le{}1\\
			0 & \ptxt{otherwise}
		\end{array}\right.
	\]
	What are $f_{X}$ and $f_{Y}$?
	\begin{align*}
		f_{X}(x) & =\int_{-\sqrt{1-x^{2}}}^{\sqrt{1-x^{2}}}\frac{1}{\pi}\,dy=\frac{2}{\pi}\sqrt{1-x^{2}}\qquad{}x\in[-1,1]\\
		f_{Y}(y) & =\int_{-\sqrt{1-y^{2}}}^{\sqrt{1-y^{2}}}\frac{1}{\pi}\,dx=\frac{2}{\pi}\sqrt{1-y^{2}}\qquad{}y\in[-1,1]
	\end{align*}
	$f_{X}(x)f_{Y}(y)=\frac{4}{\pi^{2}}\sqrt{1-x^{2}}\sqrt{1-y^{2}}\ne{}f(x,y)$ in general, so $X$ and $Y$ are not independent.\n
}

\subsection*{The Bivariate Normal Distribution}

\defn{
	$(X,Y)$ follows a \u{bivariate normal distribution} with parameters $\mu_{X},\mu_{Y}$, $\sigma_{X},\sigma_{Y}>0$, and $\rho\in(-1,1)$ if the pdf is
	\[
		f(x,y)=\frac{1}{2\pi\sigma_{X}\sigma_{Y}\sqrt{1-\rho^{2}}}e^{\displaystyle\brack{-\frac{1}{2(1-\rho^{2})}\paren{\frac{(x-\mu_{X})^{2}}{\sigma_{X}^{2}}+\frac{(y-\mu_{Y})^{2}}{\sigma_{Y}^{2}}-\frac{2\rho(x-\mu_{X})(y-\mu_{Y})}{\sigma_{X}\sigma_{Y}}}}}
	\]
}

\par\noindent
The special case where $\rho=0$ is the case where $X\perp{}Y$.\n

\par\noindent
Recall: For $2$ independent random variables $X$ and $Y$, $P(X\in{}A,Y\in{}B)=P(X\in{}A)P(Y\in{}B)$.\n
Random variables $X_{1},\ldots,X_{n}$ are independent when $f(x_{1},\ldots,x_{n})=f(x_{1})\cdots{}f(x_{n})$, $\forall{}(x_{1},\ldots,x_{n})$, which is true when $F(x_{1},\ldots,x_{n})=F(x_{1})\cdots{}F(x_{n})$, $\forall{}(x_{1},\ldots,x_{n})$.

\subsection*{Conditional Distributions}

\defn{
	For discrete random variables $X$ and $Y$, the \u{conditional probability density function} of $X$ given $Y$ is
	\[
		p_{X\mid{}Y}(x\mid{}y)=\frac{p_{X,Y}(x,y)}{p_{Y}(y)}
	\]
	when $p_{Y}(y)>0$, and $0$ otherwise. That is, $p_{X,Y}(x,y)=p_{X\mid{}Y}(x\mid{}y)p_{Y}(y)$.\n
}

\par\noindent
Consequently, the marginal distribution of $X$ is $p_{X}(x)=\sum_{y}p_{X\mid{}Y}(x\mid{}y)p_{Y}(y)$.\n

\defn{
	For continuous random variables $X$ and $Y$, the \u{conditional probability density function} of $X$ given $Y$ is
	\[
		f_{X\mid{}Y}(x\mid{}y)=\frac{f_{X,Y}(x,y)}{f_{Y}(y)}
	\]
	when $f_{Y}(y)>0$, and $0$ otherwise.\n
}

\ex{
	Say $(X,Y)$ follow a bivariant normal distribution with parameters $\mu_{X},\mu_{Y},\sigma_{X},\sigma_{Y},\rho$. The conditional density of $X$ given $Y=y$ is
	\[
		\Normal\paren{\mu_{X}+\rho\frac{\sigma_{X}}{\sigma_{Y}}(y-\mu_{Y}),(1-\rho^{2})\sigma_{X}^{2}}
	\]
	Observe that the variance of this new distribution is no more than that of $X$.\n
}

\ex{
	Say $X_{1}\sim\Unif[0,1]$, and $X_{2}\sim\Unif[0,X_{1}]$. We want to find the joint pdf of $(X_{1},X_{2})$, and the marginal pdf of $X_{2}$. Well, $f_{X_{1}}=1$ and $f_{X_{2}\mid{}X_{1}}=\frac{1}{X_{1}}$. So $f_{X_{1},X_{2}}(x_{1},x_{2})=f_{X_{1}}(x_{1})f_{X_{2}}(x_{2})=\frac{1}{x_{1}}$, for $0\le{}x_{1}\le{}1$, and $0\le{}x_{2}\le{}x_{1}$ (and $0$ otherwise). Thus,
	\[
		f_{X_{2}}(x_{2})=\int_{x_{2}}^{1}\frac{1}{x_{1}}\,dx_{1}=\brack{\ln(x_{1})}_{x_{1}=x_{2}}^{x_{1}=1}=-\ln{}x_{2}\qquad{}0<x_{2}\le{}1
	\]
}

\ex{
	(Bayesian Posterior)\n
	Given $0<\theta<1$, suppose $X\sim\Bin(n,\theta)$. Suppose the prior pdf of $\Theta$ is $\Unif[0,1]$. What is $p_{X}(x)$?\nn
	Well, $f(x,\theta)=\binom{n}{x}\theta^{x}(1-\theta)^{n-x}$ when $x=0,1,\ldots,n$, and $0<\theta<1$. So we can compute the marginal pdf of $X$:
	\begin{align*}
		p_{X}(x) & =\int_{0}^{1}\binom{n}{x}\theta^{x}(1-\theta)^{n-x}\,d\theta\\
		& =\binom{n}{x}\int_{0}^{1}\theta^{x}(1-\theta)^{n-x}\,d\theta\\
		& =\binom{n}{x}\frac{\Gamma(x+1)\Gamma(n-x+1)}{\Gamma(n+x)}\underbrace{\int_{0}^{1}\frac{\Gamma(x+1+n-x+1)}{\Gamma(x+1)\Gamma(n-x+1)}\theta^{(x-1)+1}(1-\theta)^{(n-x+1)-1}\,d\theta}_{\ptxt{The pdf of $\Beta(x+1,n-x+1)$}}\\
		& =\binom{n}{x}\frac{\Gamma(x+1)\Gamma(n-x+1)}{\Gamma(n+x)}
	\end{align*}
	Now, show that the posterior pdf of $\Theta$ given $X=x$ is
	\[
		\frac{\Gamma(n+2)}{\Gamma(x+1)\Gamma(n-x+1)}\theta^{x}(1-\theta)^{n-x}
	\]
	which is a beta distribution. Well
	\[
		f(x,\theta)=\binom{n}{x}\theta^{x}(1-\theta)^{n-x};\quad{}f_{X}(x)=p_{X}(x)=\binom{n}{x}\frac{\Gamma(x+1)\Gamma(n-x+1)}{\Gamma(n+2)}
	\]
	so
	\[
		f_{\Theta\mid{}X}(\theta)=\frac{f(x,\theta)}{f_{X}(x)}=\frac{\binom{n}{x}\theta^{x}(1-\theta)^{n-x}}{\binom{n}{x}\frac{\Gamma(x+1)\Gamma(n-x+1)}{\Gamma(n+2)}}=\frac{\Gamma(n+2)}{\Gamma(x+1)\Gamma(n-x+1)}\theta^{x}(1-\theta)^{n-x}\sim\Beta(x+1,n-x+1)
	\]
	\proven
}
