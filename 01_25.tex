\defn{
	The \u{sample space}, denoted $\Omega$, is the collection of all outcomes of an experiment or process.\n
}

\ex{
	\begin{itemize}[topsep=0pt, itemsep=0pt, leftmargin=3\parindent]
		\item Tossing a coin: $\Omega=\set{H,T}$
		\item Tossing 2 coins: $\Omega=\set{(H,H),(H,T),(T,H),(T,T)}$
		\item Rolling 2 dice and summing: $\Omega=\set{2,\ldots,12}$
	\end{itemize}\up\n
}

\defn{
	An \u{event} is a subset of $\Omega$. A \u{simple event} is a single outcome, whereas a \u{compound event} consists of more than one outcome.\n
}

\defn{
	An event $A$ is said to \u{occur} if the actual outcome of the experiment is contained in $A$.\n
}

\defn{
	A \u{discrete sample space} has a finite or countably infinite number of elements. A \u{continuous sample space} has an uncountable number of elements.\n
}

\defn{
	Two events $A$ and $B$ are said to be \u{mutually exclusive} if they're disjoint, i.e., $A\cap{}B=\emptyset$.\n
}

\par\noindent
The probability axioms:
\begin{itemize}
	\item $P(\Omega)=1$
	\item $P(A)\ge{}0$, $\forall{}A\subseteq\Omega$
	\item For a finite family of mutually exclusive events $A_{1},\ldots,A_{k}$, $P(\bigcup_{i=1}^{k}A_{i})=\sum_{i=1}^{k}P(A_{i})$.
	\item For a countably infinite family of mutually exclusive events $A_{1},\ldots$, $P(\bigcup_{i=1}^{\infty}A_{i})=\sum_{i=1}^{\infty}P(A_{i})$.
\end{itemize}

\par\noindent
In general, a compound event is the (disjoint) union of simple events.\n

\par\noindent
If $\Omega$ consists of $N$ equally likely simple events, then $\forall{}A\subseteq\Omega$, $P(A)=\frac{\card{A}}{N}$.\n

\par\noindent
Properties of probability:
\begin{itemize}
	\item $P(A\comp)=1-P(A)$
	\item $P(\emptyset)=0$
	\item $A\subseteq{}B\Rightarrow{}P(A)\le{}P(B)$
	\item $\forall{}A,B$, $P(A\cup{}B)=P(A)+P(B)-P(A\cap{}B)$
\end{itemize}

\defn{
	The probability of an event $A$, given another event $B$, is called the \u{conditional probability} of $A$ given $B$, and is denoted $P(A|B)$. It's defined by $P(A|B)P(B)=P(A\cap{}B)$.\n
}

\defn{
	Two events $A$ and $B$ are \u{independent} if $P(A|B)=P(A)$; $P(B|A)=P(B)$.\n
}

\par\noindent
If $A$ and $B$ are independent, then $P(A\cap{}B)=P(A)P(B)$.\n

\par\noindent
Consider selecting $r$ (distinguishable) items from a set of $n$.
\begin{itemize}
	\item When sampling with replacement, there are $n^{r}$ possibilities.
	\item When sampling without replacement (and consider order), there are $\frac{n!}{(n-r)!}$ possibilities.
	\item When sampling without replacement (and not considering order), there are $\binom{n}{r}=\frac{n!}{(n-r)!r!}$ possibilities.
\end{itemize}
