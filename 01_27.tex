\defn{
	A \u{permutation} is an ordered arrangement of a set of objects. A \u{combination} is an unordered set of objects.\n
}

\defn{
	$\binom{n}{k}$ is a \u{binomial coefficient}.\n
}

\[
	(a+b)^{n}=\sum_{k=0}^{n}\binom{n}{k}a^{k}b^{n-k}
\]

\defn{
	A \u{random variable} is a numerically-valued function on a sample space $\Omega$. Random variables are normally denoted with capital letters.\n
}

\ex{
	Tossing a coin. $\Omega=\set{H,T}$. $X(H)=0$, $X(T)=1$.\n
}

\par\noindent
Random variables are typically discrete or continuous (although a mixture is possible), determined by their corresponding sample space.\n

\par\noindent
Notation: The event $X=x\rightsquigarrow\set{\omega\in\Omega\mid{}X(\omega)=x}$.\n
\phantom{Notation: }The event $a\le{}X\le{}B\rightsquigarrow\set{\omega\in\Omega\mid{}a\le{}X(\omega)\le{}b}$.\n

\defn{
	The \u{probability distribution} of a discrete random variable is a list of distinct values $x_{i}$ of $X_{i}$, together with their associated probabilities, denoted
	\[
		p(x)=P(X=x)=P(\set{\omega\in\Omega\mid{}X(\omega)=x})
	\]
	Another name is the \u{probability mass function} or \u{probability density function}. It's commonly abbreviated \u{pmf} or \u{pdf}.\n
}

\defn{
	The \u{cumulative distribution function} (or \u{cdf}) of a discrete random variable is the function
	\[
		F(x)=P(X\le{}x)=\sum_{i:x_{i}\le{}x}p(x_{i})
	\]
	for $-\infty<x<\infty$.\n
}

\ex{
	Tossing a coin.\n
	\begin{tabular}{c|cc}
		s & H & T\\ \hline
		X(s) & 0 & 1
	\end{tabular}
	\n
	\begin{itemize}[topsep=0pt, itemsep=0pt, leftmargin=3\parindent]
		\item $p(0)=P(X=0)=0.5$
		\item $p(1)=P(X=1)=0.5$
		\item $F(0)=P(X\le{}0)=P(X=0)=0.5$
		\item $F(0.5)=P(X\le{}0.5)=P(X=0)=0.5$
		\item $F(1)=P(X\le{}1)=P(X=0)+P(X=1)=1$
	\end{itemize}
}

\par\noindent
Suppose we don't know if the coin is fair. Instead, we have some $0<\alpha<1$ where $p(0)=1-\alpha$, $p(1)=\alpha$. We write
\[
	p(x;\alpha)=\left\{\begin{array}{ll}
		1-\alpha & x=0\\
		\alpha & x=1\\
		0 & \ptxt{otherwise}
	\end{array}\right.
\]
Here, $\alpha$ is called a \u{parameter} of the distribution, and the collection of distributions $\set{p(x;\alpha)\mid{}\alpha\in\mathcal{A}}$ is called a \u{family} of distributions. $\mathcal{A}$ is the \u{parameter space}.\n

\defn{
	A \u{Bernoulli} random variable is a random variable whose only possible values are $0$ and $1$. We say $X\sim\Ber(p)$, and $P(X=1)=p$, $P(X=0)=1-p$, for $p\in[0,1]$.\n
}

\defn{
	A \u{Binomial} random variable is the sum of independent and identically distributed (i.i.d.) Bernoulli random variables. Given $Z_{1},\ldots,Z_{n}\sim\Ber(p)$, $X=\sum_{i=1}^{n}Z_{i}$, we say $X\sim\Bin(n;p)$. The pmf for $X$ is
	\[
		P(X=k)=\binom{n}{k}p^{k}(1-p)^{n-k}
	\]
}

\defn{
	A \u{Geometric} random variable is the number of i.i.d. Bernoulli trials before a success. If $Z_{i}\sim\Ber(p)$, we say $X\sim\Geom(p)$, and its pmf is
	\[
		P(X=k)=(1-p)^{k-1}p\qquad{}k=1,2,\ldots
	\]
}

\defn{
	A \u{Negative Binomial} random variable is the number of i.i.d. Bernoulli trials before $r$ successes (for fixed $r$). If $Z_{i}\sim\Ber(p)$, then the pmf of $X$ is
	\[
		P(X=k)=\binom{k-1}{r-1}p^{r}(1-p)^{k-r}\qquad{}k\ge{}r
	\]
}

\defn{
	A \u{Poisson} random variable counts the number of events in a specified interval or region. $\lambda$ is the parameter representing the rate. For $X\sim\Poisson(\lambda)$, the pmf is
	\[
		P(X=x)=\frac{e^{-\lambda}\lambda^{k}}{x!}
	\]
}

\par\noindent
We derive the Poisson distribution as the limit of a binomial distribution. Say $X\sim\Bin(n,p)$, and $p\to{}0$, $n\to\infty$, and $np\approx\lambda$. Then $p(x)\to\frac{e^{-\lambda}\lambda^{x}}{x!}$:
\[
	P(X=x)=\binom{n}{x}p^{x}(1-p)^{n-x}=\frac{n!}{x!(n-x)!}\paren{\frac{\lambda}{n}}^{x}\paren{1-\frac{\lambda}{n}}^{n}=\frac{\lambda^{x}}{x!}\frac{n!}{(n-x)!n^{x}}\paren{1-\frac{\lambda}{n}}^{-x}\paren{1-\frac{\lambda}{n}}^{n}
\]
Now, we take the limit
\[
	\lim_{n\to\infty}P(X=x)=\frac{\lambda^{x}}{x!}\lim_{n\to\infty}\underbrace{\frac{n(n-1)\cdots(n-x+1)}{n^{x}}\vphantom{\paren{\frac{\lambda}{n}}}}_{\to{}1}\underbrace{\paren{1-\frac{\lambda}{n}}^{-x}}_{\to{}1}\underbrace{\paren{1-\frac{\lambda}{n}}^{n}}_{\to{}e^{-\lambda}}=\frac{\lambda^{x}e^{-\lambda}}{x!}
\]
\proven
