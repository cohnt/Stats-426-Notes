\subsection*{Covariance and Correlation}

\defn{
	Suppose $X$ and $Y$ are random variables, with means $\mu_{X},\mu_{Y}$ and standard deviations $\sigma_{X},\sigma_{Y}$ (respectively). Their \u{covariance} is
	\[
		\Cov(X,Y)=E((X-\mu_{X})(Y-\mu_{Y}))
	\]
	and their \u{correlation} is
	\[
		\Corr(X,Y)=\frac{\Cov(X,Y)}{\sigma_{X}\sigma_{Y}}
	\]
}

\par\noindent
Some properties:
\begin{itemize}
	\item $\Cov(X,Y)=E(XY)-E(X)E(Y)$
	\item $\Cov(X,Y)=\Cov(Y,X)$
	\item $\Cov(X,X)=\Var(X)$
	\item $X\perp{}Y\Rightarrow\Cov(X,Y)=0$ (The converse doesn't always hold!)
\end{itemize}

\prop{
	Suppose $X_{1},\ldots,X_{n},Y_{1},\ldots,Y_{m}$ are random variables, all with finite variances. Then
	\[
		\Cov\paren{\sum_{i=1}^{n}(a_{i}+b_{i}X_{i}),\sum_{j=1}^{m}(c_{j}+d_{j}Y_{j})}=\sum_{i=1}^{n}\sum_{j=1}^{m}b_{i}d_{j}\Cov(X_{i},Y_{j})
	\]
	for any constants $a_{i},b_{i},c_{j},d_{j}$.\n
}

\par\noindent
Some special cases:
\begin{itemize}
	\item $\Var(X+Y)=\Var(X)+\Var(Y)+2\Cov(X,Y)$
	\item $\displaystyle\Var\paren{\sum_{i=1}^{n}X_{i}}=\sum_{i=1}^{n}\Var(X_{i})+\sum_{i_{1}\ne{}i_{2}}\Cov(X_{i_{1}},X_{i_{2}})$
	\item $\displaystyle\Var\paren{\sum_{i=1}^{n}(a_{i}+b_{i}X_{i})}=\sum_{i=1}^{n}b_{i}^{2}\Var(X_{i})+\sum_{i_{1}\ne{}i_{2}}b_{i_{1}}b_{i_{2}}\Cov(X_{i_{1}},X_{i_{2}})$.
	\item $\displaystyle X_{i}\perp{}X_{j},\forall{}i\ne{}j\Rightarrow\Var\paren{\sum_{i=1}^{n}X_{i}}=\sum_{i=1}^{n}\Var(X_{i})$
\end{itemize}

\par\noindent
Let $\rho=\Corr(X,Y)=\frac{\sigma_{XY}}{\sigma_{X}\sigma_{Y}}=\frac{\Cov(X,Y)}{\sigma_{X}\sigma_{Y}}$. Then
\begin{itemize}
	\item $-1\le\rho\le{}1$.\n
	$\rho\ge-1$: $\Var(\frac{X}{\sigma_{X}}+\frac{Y}{\sigma_{Y}})=\Var(\frac{X}{\sigma_{X}})+\Var(\frac{Y}{\sigma_{Y}})+2\Cov(\frac{X}{\sigma_{X}},\frac{Y}{\sigma_{Y}})=1+1+2\rho=2(1+\rho)\ge{}0\Rightarrow\rho\ge-1$\n
	$\rho\le1$: $\Var(\frac{X}{\sigma_{X}}-\frac{Y}{\sigma_{Y}})=\cdots=2(1-\rho)\ge{}0\Rightarrow\rho\le{}1$.
	\item For bivariate normally distributed random variables $X$ and $Y$, with parameters $\mu_{X},\mu_{Y},\sigma_{X},\sigma_{Y},\rho$, we have\n
	$\Cov(X,Y)=\rho\sigma_{X}\sigma_{Y}$ and $\Corr(X,Y)=\rho$.
	\item $\rho=1$ iff $\exists{}a\in\R,b>0$ \st{} $P(Y=a+bX)=1$.\nn
	Proof: $\rho=1$ iff $\Var(\frac{X}{\sigma_{X}}-\frac{Y}{\sigma_{Y}})=2(1-\rho)=0$, so let $\mu^{*}=E(\frac{X}{\sigma_{X}}-\frac{Y}{\sigma_{Y}})$. By the Chebyshev inequality, we have\n
	$P(\frac{X}{\sigma_{X}}-\frac{Y}{\sigma_{Y}}=\mu^{*})=1=P(\frac{\sigma_{Y}}{\sigma_{X}}X-Y=\sigma_{Y}\mu^{*})=P(Y=\frac{\sigma_{Y}}{\sigma_{X}}X-\sigma_{Y}\mu^{*})$. That is, $P(Y=a+bX)=1$ with $a=-\sigma_{Y}\mu^{*}$ and $b=\frac{\sigma_{Y}}{\sigma_{X}}$.\proven\up\up\nn
	\item Likewise, $\rho=-1$ iff $\exists{}a\in\R,b<0$ \st{} $P(Y=a+bX)=1$.
\end{itemize}

\subsection*{Conditional Expectation}

\defn{
	(Continuous case)
	\begin{align*}
		E(Y|X=x) & \eqdef\int_{\R}yf_{Y|X}(y|x)\,dy
		\\
		E(h(Y)|X=x) & =\int_{\R}h(y)f_{Y|X}(y|x)\,dy
		\\
		\Var(Y|X=x) & \eqdef{}E((Y-E(Y|X=x))^{2}|X=x)
	\end{align*}
	This is also defined for the discrete case, in the obvious manner.\n
}

\par\noindent
Properties:
\begin{itemize}
	\item $E(h(Y)|X)$ is a function of $X$. For a random variable $X$, it's also a random variable.
	\item $\Var(Y|X=x)=E(Y^{2}|X=x)-E(Y|X=x)^{2}$
	\item $E(Y)=E(E(Y|X))$
	\item $\Var(Y)=\Var(E(Y|X=x))+E(\Var(Y|X=x))$
\end{itemize}

\ex{
	Let $X\sim\Gamma(\lambda,\alpha)$. Given $X=x$, the conditional distribution of $Y$ is Poisson, with mean $x$. Then $E(Y|X)=X$ and $\Var(Y|X)=X$. So
	\begin{align*}
		& E(Y)=E(E(Y|X))=E(X)=\frac{\alpha}{\lambda}\\
		& \Var(Y)=\Var(E(Y|X))+E(\Var(Y|X))=\Var(X)+E(X)=\frac{\alpha}{\lambda^{2}}+\frac{\alpha}{\lambda}
	\end{align*}
	Note that $\Var(Y)\ge\Var(E(Y|X))$.\n
}

\subsection*{Prediction}

\par\noindent
It's common to use a function of $X$, $h(X)$, to predict $Y$.\n
A common criterion is ``mean squared error'' (MSE): $h(X)=E((Y-h(X))^{2})$\n

\par\noindent
$E((Y-E(Y|X))^{2})\le{}E((Y-h(X))^{2})$, that is, the predictor $E(Y|X)$ minimizes the MSE among all predictors $h(X)$.\n

\section*{Moment Generating Functions}

\defn{
	For a random variable $X$, its \u{moment generating funciton} (mgf) is $M(t)=E(e^{tX})=\int_{\R}e^{tx}f(x)\,dx$.\n
}

\par\noindent
Properties:
\begin{itemize}
	\item For $X,Y$ random variables with mgfs $M_{X},M_{Y}$, If $M_{X}(t)=M_{Y}(t)$ for $t_{1}<t<t_{2}$, with $0\in(t_{1},t_{2})$, then $X$ and $Y$ have the same distribution.
	\item $Y=a+bX\Rightarrow{}M_{Y}(t)=e^{ta}M_{X}(tb)$
	\item $X\perp{}Y\Rightarrow{}M_{X+Y}(t)=M_{X}(t)M_{Y}(t)$, $\forall{}t$
	\item $E(X^{r})=M^{(r)}(0)$ if $M^{(r)}(0)$ exists
\end{itemize}
