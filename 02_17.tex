\ex{
	Suppose $X\sim\Normal(0,1)$. The mgf of $X$ is
	\[
		M_{X}(t)=\frac{1}{\sqrt{2\pi}}\int_{\R}e^{tx}e^{-x^{2}/2}\,dx=\frac{1}{\sqrt{2\pi}}\int_{\R}e^{-(x-t)^{2}/2}e^{t^{2}/2}\,dx=e^{t^{2}/2}
	\]
	The mgf of $Y=\sigma{}X+\mu$ is $M_{Y}(t)=e^{(\sigma{}t)^{2}/2+\mu{}t}$.\n
	$M_{Y}^{(1)}(0)=\mu$; $M_{Y}^{(2)}(0)=\mu^{2}+\sigma^{2}$.\n
}

\ex{
	If $X_{1}\perp{}X_{2}$, $X_{1}\sim\Normal(\mu_{1},\sigma)$, $X_{2}\sim\Normal(\mu_{2},\sigma)$, then
	\[
		M_{X_{1}+X_{2}}(t)\overset{X_{1}\perp{}X_{2}}{=}M_{X_{1}}(t)M_{X_{2}}(t)=e^{\mu_{1}t}e^{(\sigma{}t)^{2}/2}e^{\mu_{2}t}e^{(\sigma{}t)^{2}/2}=e^{(\mu_{1}+\mu_{2})t}e^{(\sqrt{2}\sigma{}t)^{2}/2}
	\]
	So $X_{1}+X_{2}\sim\Normal(\mu_{1}+\mu_{2},2\sigma^{2})$.\n
}

\section*{Limit Theorems}

\par\noindent
We are interested in learning the limiting behavior of the sum (or the average) of independent random variables when the number of summands becomes large.\n

\par\noindent
Consider $X_{1},\ldots,X_{n}$ i.i.d. with mean $\mu$ and standard deviation $\sigma$, with $\bar{X}_{n}=n\inv\sum_{i}X_{i}$. Recall:
\begin{itemize}
	\item $E(\bar{X}_{n})=\mu$, $\Var(\bar{X}_{n})=\sigma^{2}/n$
	\item $\Var\to{}0$ as $n\to\infty$
	\item The pdf of $\bar{X}_{n}$ has mass concentrated around $\mu$
\end{itemize}
We will discuss the weak law of large numbers (WLLN) and the central limit theorem (CLT).\n

\subsection*{Weak Law of Large Numbers}

\thm{
	(WLLN) Consider $X_{1},\ldots,X_{n}$ i.i.d., with mean $\mu$ and variance $\sigma^{2}$. Let $\bar{X}_{n}=n\inv\sum_{i}X_{i}$.\n
	Then $\forall\varepsilon>0$, $P(\abs{\bar{X}_{n}-\mu}>\varepsilon)\to{}0$ as $n\to\infty$.\nn
	Proof: This is a direct result of the Chebyshev inequality.
	\[
		P(\abs{\bar{X}_{n}-\mu}>\varepsilon)\le\frac{\Var(\bar{X}_{n})}{\varepsilon^{2}}=\frac{\sigma^{2}}{n\varepsilon^{2}}\to{}0\;\ptxt{as}\;n\to\infty
	\]
	\proven
}

\par\noindent
We also denote this by $\bar{X}_{n}\overset{P}{\to}\mu$.\n

\ex{
	Suppose $Z_{1},\ldots,Z_{n}\sim\Ber(p)$ are i.i.d., with $p\in(0,1)$. Then $Y_{n}=\sum_{i}Z_{i}\sim\Bin(n,p)$.\n
	$EZ_{i}=p$. $\frac{Y_{n}}{n}\overset{P}{\to}p$.\n
}

\ex{
	Suppose $Z_{1},\ldots,Z_{n}\sim\Exp(\lambda)$ are i.i.d. Then $Y_{n}=\sum_{i}Z_{i}\sim\DistGamma(\lambda,n)$.\n
	$EZ_{i}=\frac{1}{\lambda}$. $\frac{Y_{n}}{n}\overset{P}{\to}\frac{1}{\lambda}$.\n
}

\subsection*{Convergence in Distribution}

\defn{
	Let $Y_{1},\ldots,Y_{n}$ be a sequence of random variables, with cdfs $F_{i}$. Let $Y$ be a random variable with cdf $F$. We say that $Y_{n}$ \u{converges in distribution} to $Y$ if
	\[
		\lim_{n\to\infty}F_{n}(x)=F(x)
	\]
	at every point $x$ at which $F$ is continuous. We denote this by $Y_{n}\overset{D}{\to}Y$.\n
}

\thm{
	(Continuity Theorem) Let $F$ be a cdf with mgf $M$. If $M_{n}(t)\to{}M(t)$ for all $t$ in an open interval containing $0$, then $F_{n}(x)\to{}F(x)$ at all points $x$ for which $F$ is continuous.\n
}

\ex{
	Suppose $Z_{1},\ldots,Z_{n}\sim\Exp(\lambda)$ are i.i.d., then $X=\sum_{i}Z_{i}\sim\Gamma(\lambda,n)$. $M_{X}(t)$ is $(1-\lambda\inv{}t)^{-n}$. Let $Y=\frac{\lambda}{\sqrt{n}}X-\sqrt{n}$. What does the mgf of $Y$ converge to as $n\to\infty$? Well,
	\[
		M_{Y}(t)=e^{-\sqrt{n}t}(1-(\lambda\inv\lambda/\sqrt{n})t)^{-n}=e^{-\sqrt{n}t}(1-t/\sqrt{n})^{-n}
	\]
	And
	\begin{align*}
		\ln(M_{Y}(t)) & =-\sqrt{n}t-n\log(1-\tfrac{t}{\sqrt{n}})\\
		& =-\sqrt{n}t-n(-t/\sqrt{n}-(1/2)t^{2}/n-O(n^{-3/2}))\\
		& =\tfrac{1}{2}t^{2}-O(n^{-1/2})\\
		& \to(1/2)t^{2}\;\ptxt{as}\;n\to\infty
	\end{align*}
	Thus, $\lim_{n\to\infty}M_{Y}(t)=e^{t^{2}/2}$.\n
}

\ex{
	Suppose $Z_{1},\ldots,Z_{n}\sim\Exp(\lambda)$ are i.i.d., then $X=\sum_{i}Z_{i}\sim\Gamma(\lambda,n)$. We want to show $M_{X}(t)=(1-\lambda\inv{}t)^{-n}$. Well,
	\[
		M_{Z}(t)=E(e^{tZ})=\int_{0}^{\infty}e^{tz}\lambda{}e^{-\lambda{}z}\,dz=\lambda\int_{0}^{\infty}\frac{\lambda-t}{\lambda-t}e^{-(\lambda-t)z}\,dz=\frac{\lambda}{\lambda-t}=(1-\tfrac{t}{\lambda})^{-n}
	\]
	Because $X=\sum_{i}Z_{i}$,
	\[
		M_{X}(t)=((1-\tfrac{t}{\lambda})\inv)^{-n}=(1-t\lambda\inv)^{-n}
	\]
	Also,
	\[
		E(X)=\restr{M_{X}'(t)}_{t=0}=\restr{-n(1-\tfrac{t}{\lambda})^{-n-1}(-\tfrac{1}{\lambda})}_{t=0}=\tfrac{n}{\lambda}
	\]
}
